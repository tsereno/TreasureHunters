const canvas = document.getElementById("renderCanvas"); const engine = new BABYLON.Engine(canvas, true); const scene = new BABYLON.Scene(engine); // Create a camera const camera = new BABYLON.ArcRotateCamera("camera", -Math.PI / 2, Math.PI / 2.5, 15, new BABYLON.Vector3(0, 0, 0), scene); camera.attachControl(canvas, true); // Create a light const light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene); // Create a video texture const videoElement = document.getElementById("inputVideo"); const videoTexture = new BABYLON.VideoTexture("video", videoElement, scene, true); // Create a plane to display the video const videoPlane = BABYLON.MeshBuilder.CreatePlane("videoPlane", {width: 16, height: 9}, scene); videoPlane.position.z = 10; const videoMaterial = new BABYLON.StandardMaterial("videoMaterial", scene); videoMaterial.diffuseTexture = videoTexture; videoMaterial.emissiveColor = new BABYLON.Color3(1, 1, 1); videoPlane.material = videoMaterial; // Create a simple 3D object (a sphere) const sphere = BABYLON.MeshBuilder.CreateSphere("sphere", {diameter: 2}, scene); sphere.position.z = 0; // Position the sphere at the center // Add material to the sphere const sphereMaterial = new BABYLON.StandardMaterial("sphereMaterial", scene); sphereMaterial.diffuseColor = new BABYLON.Color3(1, 0, 0); // Red color sphereMaterial.alpha = 0.7; // Make it slightly transparent sphere.material = sphereMaterial; // Create a 2D canvas for drawing bounding boxes const objectDetectionCanvas = document.createElement('canvas'); objectDetectionCanvas.width = videoElement.width; objectDetectionCanvas.height = videoElement.height; const ctx = objectDetectionCanvas.getContext('2d'); // Create a dynamic texture for the object detection overlay const objectDetectionTexture = new BABYLON.DynamicTexture("objectDetectionTexture", {width: videoElement.width, height: videoElement.height}, scene); const objectDetectionMaterial = new BABYLON.StandardMaterial("objectDetectionMaterial", scene); objectDetectionMaterial.diffuseTexture = objectDetectionTexture; objectDetectionMaterial.specularColor = new BABYLON.Color3(0, 0, 0); objectDetectionMaterial.emissiveColor = new BABYLON.Color3(1, 1, 1); objectDetectionMaterial.alpha = 0.7; // Create a plane for the object detection overlay const objectDetectionPlane = BABYLON.MeshBuilder.CreatePlane("objectDetectionPlane", {width: 16, height: 9}, scene); objectDetectionPlane.position.z = 10.01; // Slightly in front of the video plane objectDetectionPlane.material = objectDetectionMaterial; // Start the webcam and object detection let objectDetector; const initializeObjectDetection = async () => { const vision = await FilesetResolver.forVisionTasks("https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision@latest/wasm"); objectDetector = await ObjectDetector.createFromOptions(vision, { baseOptions: { modelAssetPath: "https://storage.googleapis.com/mediapipe-models/object_detector/efficientdet_lite0/float16/1/efficientdet_lite0.tflite", delegate: "GPU" }, scoreThreshold: 0.5, maxResults: 5 }); navigator.mediaDevices.getUserMedia({ video: true }) .then(function(stream) { videoElement.srcObject = stream; videoElement.play(); detectObjects(); }) .catch(function(err) { console.log("An error occurred: " + err); }); }; const detectObjects = async () => { if (videoElement.readyState === 4) { const detections = await objectDetector.detect(videoElement); drawDetections(detections); } requestAnimationFrame(detectObjects); }; const drawDetections = (detections) => { ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height); for (const detection of detections) { const [x, y, width, height] = detection.boundingBox; ctx.strokeStyle = "#00FF00"; ctx.lineWidth = 2; ctx.strokeRect(x, y, width, height); ctx.fillStyle = "#00FF00"; ctx.font = "18px Arial"; ctx.fillText(detection.categories[0].categoryName, x, y > 20 ? y - 5 : y + 20); } objectDetectionTexture.update(objectDetectionCanvas); }; // Virtual Joystick let leftJoystick = new BABYLON.VirtualJoystick(true); let rightJoystick = new BABYLON.VirtualJoystick(false); // Joystick control for the sphere scene.onBeforeRenderObservable.add(() => { if (leftJoystick.pressed) { sphere.position.x += leftJoystick.deltaPosition.x * 0.1; sphere.position.z -= leftJoystick.deltaPosition.y * 0.1; } if (rightJoystick.pressed) { sphere.position.y += rightJoystick.deltaPosition.y * 0.1; } }); // Set up collision scene.collisionsEnabled = true; sphere.checkCollisions = true; videoPlane.checkCollisions = true; // Initialize object detection initializeObjectDetection(); // Render loop engine.runRenderLoop(function () { scene.render(); }); // Resize event window.addEventListener("resize", function () { engine.resize(); });
